import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int H = sc.nextInt();
		int M = sc.nextInt();
		
		// 45분 전에 일어나서 45분 자니,,,?
		if(M < 45) {
			H--;
			M = 60-(45-M);
			if(H < 24) {
				H = 24-H; // 어떻게 시도해야하는지,,,?
			}
		}else {
			M = M-45;
		}
		System.out.println(H + " " + M);
	}

}
